Tailwind CSS (Utility classes)

-- Background Colors --
-bg-{color}-
To change the background color : bg-red-400

-Arbitary colors-
To setup the background color non default : bg-[#f254g1]

-Opacity-
To add the opacity level : bg-red-400/70

-Gradiants-
To add a gradiant : bg-gradiant-to-tl (tl:top left; r:right; tr:top righ;)
                    from-green-400    (Starting color of the gradiant)
                    via-blue-200      (The mid range color of the gradiant)
                    to-red-300        (The end colr of the gradiant)



-- Width & Height --
-width-
To change the width : w-30 

-height-
To change the height : h-40

-Arbitary value-
To add an arbitary value as height or width : w-[200px]  h-[50vh]

-Percentage-
To add a percentage as height or width : w-1/4  (25%)
                                        w-3/4  (75%)

-Max width/height-
To add max width : max-w-xl (xl is the value for the screen : md,s)

-Max width screen-
This helps us to give the max width of an element : max-w-screen-xl (elements width will be xl)

const SignUp = () => {
  const navigate = useNavigate();
  const [role, setRole] = useState('counselor'); // default selected role

  return (
    <div className="mx-auto w-full max-w-md min-h-screen bg-white shadow-md rounded-2xl p-6 md:p-10">
      {/* Logo */}

      {/* Back to home */}
      <div className="hidden md:flex items-center mt-4 text-sm text-gray-600 font-semibold">

          Home
        </Link>
      </div>

      {/* Role Tabs */}
      <div className="flex bg-gray-200 p-1 rounded-full mt-6">

          Student

        <button
          onClick={() => setRole('counselor')}

          Counselor
        </button>
      </div>

      {/* Form Header */}
      {role === 'counselor' && (
        <>
          <h2 className="text-2xl font-bold text-center mt-6">Apply as Counselor</h2>

            </div>

            <div>
              <label className="block text-sm font-medium mb-1">Email</label>
              <input
                type="email"
                placeholder="Enter your email"

            </div>
